#!/bin/bash

set -e

echo "========================================"
echo " DevOps Lab Setup - Amazon Linux 2023"
echo "========================================"

# Update System
echo "Updating system..."
dnf update -y

# Install Basic Packages
echo "Installing basic packages..."
dnf install -y \
git \
python3 \
python3-pip \
docker \
wget \
curl \
tar \
make \
zip \
unzip \
jq \
awscli \
java-21-amazon-corretto

# Configure Git
echo "Configuring Git..."
git config --global user.name "farhanshaikh-cloud"
git config --global user.email "shaikhfarhan7644@gmail.com"

echo "Git Configuration:"
git config --global --list

# Docker
echo "Starting Docker..."
systemctl enable docker
systemctl start docker

# Add current user to Docker group
usermod -aG docker ec2-user

#################################################
# Jenkins Installation
#################################################

echo "Installing Jenkins..."

wget -O /etc/yum.repos.d/jenkins.repo \
https://pkg.jenkins.io/redhat-stable/jenkins.repo

rpm --import https://pkg.jenkins.io/redhat-stable/jenkins.io-2023.key

dnf install -y jenkins

systemctl enable jenkins
systemctl start jenkins

#################################################
# Terraform Installation
#################################################

echo "Installing Terraform..."

dnf install -y dnf-plugins-core

dnf config-manager addrepo \
--from-repofile=https://rpm.releases.hashicorp.com/AmazonLinux/hashicorp.repo

dnf install -y terraform

#################################################
# Grafana Installation
#################################################

echo "Installing Grafana..."

cat <<EOF >/etc/yum.repos.d/grafana.repo
[grafana]
name=Grafana
baseurl=https://rpm.grafana.com
repo_gpgcheck=1
enabled=1
gpgcheck=1
gpgkey=https://rpm.grafana.com/gpg.key
sslverify=1
EOF

dnf install -y grafana-enterprise

systemctl enable grafana-server
systemctl start grafana-server

#################################################
# Ansible Installation
#################################################

echo "Installing Ansible..."

dnf install -y ansible

#################################################
# Versions
#################################################

echo
echo "========================================"
echo "Installed Versions"
echo "========================================"

git --version
python3 --version
pip3 --version
docker --version
java -version
terraform -version
aws --version
ansible --version

echo
echo "========================================"
echo "Services Status"
echo "========================================"

systemctl --no-pager --full status docker | head -10
systemctl --no-pager --full status jenkins | head -10
systemctl --no-pager --full status grafana-server | head -10

echo
echo "========================================"
echo "Jenkins Initial Password"
echo "========================================"

cat /var/lib/jenkins/secrets/initialAdminPassword

echo
echo "========================================"
echo "DevOps Lab Setup Completed Successfully!"
echo "========================================"

echo
echo "NOTE:"
echo "1. Logout and login again for Docker group changes."
echo "2. Jenkins  : http://<EC2-PUBLIC-IP>:8080"
echo "3. Grafana  : http://<EC2-PUBLIC-IP>:3000"
echo "4. Default Grafana credentials:"
echo "   Username: admin"
echo "   Password: admin"
